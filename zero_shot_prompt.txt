def to_snake_case(text):
	"""
	Convert a string to snake_case format.

	This function takes an input string and returns a new string in snake_case format. Words are separated by underscores, and all letters are lowercase. Non-alphanumeric characters are replaced with underscores, and leading/trailing underscores are removed.

	Args:
		text (str): The input string to convert. Can contain spaces, hyphens, or other non-alphanumeric separators.

	Returns:
		str: The snake_case version of the input string.

	Examples:
		>>> to_snake_case('First Name')
		'first_name'
		>>> to_snake_case('myVariableName')
		'my_variable_name'
		>>> to_snake_case('Hello world!')
		'hello_world'
	"""
	import re
	text = re.sub(r'([a-z0-9])([A-Z])', r'\1_\2', text)
	text = re.sub(r'[^a-zA-Z0-9]+', '_', text)
	return text.lower().strip('_')
